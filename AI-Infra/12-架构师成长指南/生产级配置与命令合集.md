## 1. 系统内核全局调优

```bash
# 关闭 Swap（AI 集群必须）
swapoff -a
sed -i 's/^UUID.*swap/#&/' /etc/fstab

# 内存调优
echo 0 > /proc/sys/vm/swappiness
echo 1 > /proc/sys/vm/zone_reclaim_mode
echo 262144 > /proc/sys/vm/min_free_kbytes

# 大页内存配置
echo 524288 > /proc/sys/vm/nr_hugepages

# 网络全局调优
cat >> /etc/sysctl.conf << EOF
net.core.somaxconn = 65535
net.ipv4.tcp_tw_reuse = 1
net.core.rmem_max = 16777216
net.core.wmem_max = 16777216
net.ipv4.tcp_rmem = 4096 87380 16777216
net.ipv4.tcp_wmem = 4096 65536 16777216
EOF
sysctl -p
```

## 2. NCCL 生产环境变量

```bash
export NCCL_IB_DISABLE=0          # 启用 InfiniBand
export NCCL_P2P_DISABLE=0         # 启用 P2P（NVLink/PCIe）
export NCCL_NET_GDR_LEVEL=5       # GPUDirect RDMA 最高级别
export NCCL_DEBUG=WARN            # 日志级别（INFO/WARN/TRACE）
export NCCL_THREADS=8             # 通信线程数
export NCCL_TOPO_FILE=/path/to/topo.xml  # 自定义拓扑（可选）
```

## 3. 硬件与网络排查

```bash
# GPU 拓扑与健康
nvidia-smi topo -m                 # GPU 拓扑矩阵
dcgmi health status                # GPU 健康状态（需 DCGM）

# 高速网络
ibstat                             # InfiniBand 状态
ibv_devinfo                        # IB 设备详细信息
rdma link show                     # RDMA 链路状态

# NUMA 与 CPU
numastat                           # NUMA 内存统计
lscpu -e                           # CPU 拓扑详情

# 内核日志
dmesg | tail -100                  # 最近内核日志
dmesg | grep -i gpu                # GPU 相关内核错误
dmesg | grep -i nccl               # NCCL 相关错误
```

## 7. 常用排查方法论

### GPU 掉卡

```bash
nvidia-smi                   # 检查是否所有 GPU 可见
dmesg | grep -i "NVRM"      # 检查驱动日志
lspci | grep -i nvidia       # 检查 PCIe 设备
```

### NCCL 通信超时

```bash
# 逐步增加日志级别
NCCL_DEBUG=INFO python train.py
NCCL_DEBUG=TRACE python train.py  # 详细日志

# 检查网络连通性
ib_send_bw -d mlx5_0         # RDMA 带宽测试
```

### 存储 IO 瓶颈

```bash
iostat -x 1                   # 磁盘 IO 实时监控
iotop                         # 进程级 IO 排序
```
